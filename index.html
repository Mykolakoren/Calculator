<script>

function fmt(n) {

  return parseFloat(n).toFixed(2);

}

 

function calc() {

  var d = parseInt(document.getElementById('days').value) || 1;

  var c = parseInt(document.getElementById('chickens').value) || 1;

  var w = parseInt(document.getElementById('whole').value) || 0;

 

  var wt = parseFloat(document.getElementById('weight').value) || 1;

  var ck = parseFloat(document.getElementById('costkg').value) || 0;

  var pw = parseFloat(document.getElementById('pricewhole').value) || 0;

  var pp = parseFloat(document.getElementById('priceparts').value) || 0;

 

  var r = parseFloat(document.getElementById('rent').value) || 0;

  var u = parseFloat(document.getElementById('util').value) || 0;

  var f = parseFloat(document.getElementById('fot').value) || 0;

  var o = parseFloat(document.getElementById('other').value) || 0;

  var cap = parseFloat(document.getElementById('capex').value) || 0;

 

  var mc = d * c;

  var mk = mc * wt;

  var pc = mk * ck;

 

  var wc = mc * w / 100;

  var pc2 = mc - wc;

 

  var r1 = wc * wt * pw;

  var r2 = pc2 * wt * pp;

  var rt = r1 + r2;

 

  var fc = r + u + f + o;

  var tc = fc + pc;

  var pr = rt - tc;

 

  var marg = rt > 0 ? ((rt - pc) / rt * 100).toFixed(1) : 0;

 

  var ppc = wt * (pw + pp) / 2 - ck * wt;

  var bec = ppc > 0 ? fc / ppc : 0;

  var bed = Math.ceil(bec / c);

 

  var roi = pc > 0 ? ((pr / pc) * 100).toFixed(1) : 0;

  var dp = (pr / d).toFixed(2);

  var pch = mc > 0 ? (pc / mc).toFixed(2) : 0;

 

  var cm = pr > 0 ? Math.ceil(cap / pr) : 999;

 

  document.getElementById('rev1').textContent = fmt(r1);

  document.getElementById('rev2').textContent = fmt(r2);

  document.getElementById('revtotal').textContent = fmt(rt);

 

  document.getElementById('cost1').textContent = fmt(fc);

  document.getElementById('cost2').textContent = fmt(pc);

  document.getElementById('costtotal').textContent = fmt(tc);

 

  document.getElementById('profit').textContent = fmt(pr);

  document.getElementById('daily').textContent = fmt(dp);

  document.getElementById('margin').textContent = marg + '%';

  document.getElementById('breakeven').textContent = bed;

  document.getElementById('roi').textContent = roi + '%';

 

  document.getElementById('monthlychickens').textContent = mc;

  document.getElementById('monthlykg').textContent = fmt(mk);

  document.getElementById('perchicken').textContent = fmt(pch);

 

  if (cap > 0) {

  document.getElementById('capexbox').style.display = 'block';

  document.getElementById('capexmonths').textContent = cm;

  document.getElementById('capexprofit').textContent = fmt(pr);

  } else {

  document.getElementById('capexbox').style.display = 'none';

  }

 

  var pb = document.getElementById('profitbox');

  if (pr >= 0) {

  pb.className = 'box box-green';

  } else {

  pb.className = 'box box-red';

  }

}

 

var inps = document.querySelectorAll('input');

for (var i = 0; i < inps.length; i++) {

  inps[i].onchange = calc;

  inps[i].oninput = calc;

}

 

calc();

 

if ('serviceWorker' in navigator) {

  navigator.serviceWorker.register('data:text/javascript,self.addEventListener("install",e=>self.skipWaiting());self.addEventListener("activate",e=>e.waitUntil(clients.claim()));');

}
